# Get version of ruby
```
# Get global version
rbenv global
# Get local version
rbenv local
```

# install a Ruby version:
rbenv install 3.1.2

# choose local Ruby version 3.1.2:
rbenv local 3.1.2

# choose global Ruby version 3.1.2:
rbenv global 3.1.2
rbenv global system

# system version
/usr/bin/ruby -v

# local version 
cat .ruby-version
